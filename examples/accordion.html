JavaScript Implementation

<script>
// Toggle individual accordion item (allows multiple open)
function toggleAccordion(itemId) {
  const content = document.getElementById(itemId);
  const icon = document.getElementById('icon-' + itemId);

  content.classList.toggle('open');
  icon.classList.toggle('rotate');
}

// Toggle accordion with single open at a time
function toggleAccordionSingle(groupName, itemId) {
  const allItems = document.querySelectorAll(`[data-group="${groupName}"]`);
  const content = document.getElementById(itemId);
  const icon = document.getElementById('icon-' + itemId);

  // Close all items in the group
  allItems.forEach(item => {
    if (item.id !== itemId && item.classList.contains('open')) {
      item.classList.remove('open');
      const itemIcon = document.getElementById('icon-' + item.id);
      if (itemIcon) itemIcon.classList.remove('rotate');
    }
  });

  // Toggle the clicked item
  content.classList.toggle('open');
  icon.classList.toggle('rotate');
}
</script>

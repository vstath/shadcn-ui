Interactive Demo

<script>
function toggleContent() {
  const container = document.getElementById('content-container');
  const isLoading = container.querySelector('.animate-pulse') !== null;

  if (isLoading) {
    // Show actual content
    container.innerHTML = `
      <div class="flex items-center space-x-4">
        <img src="..." class="h-12 w-12 rounded-full" />
        <div>
          <h3 class="font-semibold">John Doe</h3>
          <p class="text-sm text-muted-foreground">Software Developer</p>
        </div>
      </div>
      <p class="mt-4 text-sm">Actual content loaded!</p>
    `;
  } else {
    // Show skeleton
    container.innerHTML = '...'; // skeleton HTML
  }
}
</script>

Without Animation

Simply remove the animate-pulse class for a static skeleton.
